# Alle regels staan bewust in één "User-agent: *"-groep. Een crawler volgt

Disallow: /dj$

# --- Niet indexeren: tracking-parameters (voorkomt duplicate content) ---
Disallow: /*?utm_
Disallow: /*&utm_
Disallow: /*?fbclid=

Disallow: /*?msclkid=
Disallow: /*?ref=

# zekerheid zodat ze nooit per ongeluk geïndexeerd raken.
Disallow: /src/
Disallow: /node_modules/
Disallow: /dist/
Disallow: /.git/
Disallow: /api/
